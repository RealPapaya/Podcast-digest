## 🚀 使用方式

### 基本用法（個股分析）：
```python
from src.render import render_email_html
from src.stock_data import enrich_stocks_with_data

digest = {
    "ep_number": "EP654",
    "stocks": [
        {"name": "台積電", "ticker": "2330", ...}
    ]
}

# 獲取股價
digest['stocks'] = enrich_stocks_with_data(digest['stocks'])

# 渲染HTML
html = render_email_html(digest)
```

### 進階用法（族群分析）：
```python
digest = {
    "ep_number": "EP654",
    "sector_analysis": [
        {
            "sector_name": "被動元件",
            "stance": "看多",
            "related_stocks": [
                {"name": "國巨", "ticker": "2327", "exchange": "台股"},
                {"name": "華新科", "ticker": "2492", "exchange": "台股"}
            ],
            ...
        }
    ]
}

# 獲取族群股票股價
for sector in digest['sector_analysis']:
    sector['related_stocks'] = enrich_stocks_with_data(sector['related_stocks'])

# 渲染HTML
html = render_email_html(digest)
```
